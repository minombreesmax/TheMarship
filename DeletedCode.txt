    void RocksSpawn()
    {
        if(DataHolder.fly)
        {
            GameObject rock = Rocks[UnityEngine.Random.Range(0, Rocks.Length)];
            var position = new Vector3(PositionX[UnityEngine.Random.Range(0, PositionX.Length)], 5f, 0f);
            var rotation = new Vector3(0f, RotationY[UnityEngine.Random.Range(0, RotationY.Length)], 0f);
            Instantiate(rock, position, Quaternion.Euler(rotation));
        }
    }

    void CraterSpawn()
    {
        GameObject crater = Craters[UnityEngine.Random.Range(0, Craters.Length)];
        var position = new Vector3(UnityEngine.Random.Range(80f, 100f), 5f, UnityEngine.Random.Range(-30f, 20f));
        Instantiate(crater, position, Quaternion.identity);
    }

    void MeteorSpawn()
    {
        if(DataHolder.fly)
        {
            GameObject meteor = Meteors[UnityEngine.Random.Range(0, Craters.Length)];
            var position = new Vector3(UnityEngine.Random.Range(90f, 130f), UnityEngine.Random.Range(15f, 30f), 0f);
            Instantiate(meteor, position, Quaternion.identity);
        }
    }
